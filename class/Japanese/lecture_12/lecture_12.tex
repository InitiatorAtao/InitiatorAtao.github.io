    \section{助词一览}
    \begin{itemize}
        \item は
            \begin{enumerate}
                \item 主语
            \end{enumerate}
        \item の
            \begin{enumerate}
                \item 的
            \end{enumerate}
        \item か
            \begin{enumerate}
                \item 疑问
            \end{enumerate}
        \item に
            \begin{enumerate}
                \item 存在场所
                \item 具体时间
                \item 基准\sn{一般后接いい或悪い(わるい)表示好坏,或遠い(とおい)或近い(ちかい)表示远近}
                \item 比例
                \item 对象
                \item 移动目的地
                \item 变化结果
                \item 动作的目的
                \item 归着点
                \item 变化后果
            \end{enumerate}
        \item が
            \begin{enumerate}
                \item 小主语
                \item 表转折
            \end{enumerate}
        \item と
            \begin{enumerate}
                \item 和
            \end{enumerate}
        \item や
            \begin{enumerate}
                \item 一类
            \end{enumerate}
        \item も
            \begin{enumerate}
                \item 也\sn{可以替代其他助词}
            \end{enumerate}
        \item より
            \begin{enumerate}
                \item 比
            \end{enumerate}
        \item で
            \begin{enumerate}
                \item 范围\sn{相当于"在"}
                \item 动作场所
                \item 动作方法,手段
            \end{enumerate}
        \item から
            \begin{enumerate}
                \item 起点\sn{相当于"从...(开始)"}
            \end{enumerate}
        \item まで
            \begin{enumerate}
                \item 终点\sn{相当于"到...(为止)"}
            \end{enumerate}
        \item を
            \begin{enumerate}
                \item 宾语
            \end{enumerate}
        \item ね
            \begin{enumerate}
                \item 确认,叮嘱,感叹
            \end{enumerate}
        \item よ
            \begin{enumerate}
                \item 判断,主张,提醒
            \end{enumerate}
        \item へ
            \begin{enumerate}
                \item 动作方向\sn{相当于"向...,往..."}
            \end{enumerate}
        \item ので
            \begin{enumerate}
                \item 因果关系\sn{相当于"因为...所以..."}
            \end{enumerate}
        \item として
            \begin{enumerate}
                \item 身份,资格,立场\sn{相当于"作为"}
            \end{enumerate}
        \item あまり
            \begin{enumerate}
                \item 不太...\sn{后接用言否定式}
            \end{enumerate}
        \item くらい(ぐらい)
            \begin{enumerate}
                \item 表示大概
            \end{enumerate}
        \item 欲しい(ほしい)
            \begin{enumerate}
                \item 想要...
            \end{enumerate}
    \end{itemize}
