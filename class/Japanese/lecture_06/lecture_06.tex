    \section{第六课-両親は名古屋にいます}
    \begin{definition}
        (存在句) 表示某物存在于某处,句尾あります用于非生物,います用于生物
        \begin{itemize}
            \item ...に(は)...があります(います) : ...处有...(某处有某物)
            \item ...は...にあります(います) : ...在...(某物在某处)
        \end{itemize}
    \end{definition}
    \subsection{助词一览}
    \begin{itemize}
        \item に 表示存在的场所
        \item が 表示存在的对象
        \item と 连接名词,表示"和"
        \item や 连接名词,表示"一类"
        \item も 替代其他助词,表示"也"
    \end{itemize}
